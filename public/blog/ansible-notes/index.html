Ansible playbook format:#

- hosts: all
  user: ubuntu
  sudo: yes
  vars:
    env: sandbox
    ver: "0.1.0"
  tasks:
    - ...
  handlers:
    - ...

Adding user:#

- name: add user jeff
  user:
    state: present
    name: jeff
    system: no
    shell: /bin/sh
    groups: sudo,docker
    append: yes
    comment: "jeff is a system admin from 1.01.2024"
    password: temp_pass

Upgrading Ubuntu system using apt:#

- name: upgrade system
  apt:
    upgrade: yes

Installing packages using apt:#

- name: install required packages on new system
  apt:
    pkg:
     - docker.io
     - docker-compose
     - nfs-common
     - ufw
     - fail2ban
    state: present
    update_cache: yes
    install_recommends: yes

Cleanup Ubuntu using apt :#

- name: system cleanup
  apt:
    autoremove: yes
    purge: yes

Pulling git repo:#

- name: pull git repo
  git:
    repo: https://github.com/Hermanowicz/my-repo.git
    dest: /src/my-app
    single_brach: yes
    version: master

Starting fail2ban service:#

- name: start fail2ban service:
  service:
    name: fail2ban
    enabled: yes
    state: started
