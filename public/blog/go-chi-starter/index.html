Golang Chi framewark starter

Starter template for Chi freamwork in golang, with make file and all nessacery commands to run new web development project.

Commands:#

# init go module in current dir
go mod init github.com/Hermanowicz/new-app

# install chi freamowk
go get -u github.com/go-chi/chi/v5

# run tidy
go mod tidy

Makefile:#

build:
  go build -o app dist/main.go

clanup:
  rm dist/app

main.go#

package main

import (
    "net/http"

    "github.com/go-chi/chi/v5"
    "github.com/go-chi/chi/v5/middleware"
)

func main() {
    r := chi.NewRouter()

    r.Use(middleware.Logger)
    r.Use(middleware.Recoverer)
    r.Use(middleware.CleanPath)
    r.Use(middleware.Heartbeat("/ping"))

    r.Get("/", func(w http.ResponseWriter, r *http.Request) {
        w.Write([]byte("Hello World!"))
    })
    http.ListenAndServe(":3000", r)
}
